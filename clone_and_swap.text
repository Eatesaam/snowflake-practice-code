// Clone
CREATE OR REPLACE DATABASE COPY_DB_CLONE
CLONE COPY_DB;

CREATE OR REPLACE SCHEMA COPY_DB_CLONE.EXTERNAL_STAGES_CLONE
CLONE MANAGE_DB.EXTERNAL_STAGES;

CREATE OR REPLACE TABLE COPY_DB_CLONE.PUBLIC.ORDERS_CLONE
CLONE COPY_DB.PUBLIC.ORDERS;

// Swap
CREATE OR REPLACE TABLE COPY_DB_CLONE.PUBLIC.ORDERS_SWAP(
    ID INT
);

ALTER TABLE COPY_DB_CLONE.PUBLIC.ORDERS_SWAP
SWAP WITH COPY_DB_CLONE.PUBLIC.ORDERS;

SELECT * FROM COPY_DB_CLONE.PUBLIC.ORDERS_SWAP;
